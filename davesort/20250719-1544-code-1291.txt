\n#!/bin/bash\n\nset -e\n\n# 1. Ensure ~/bin exists and is in PATH\nmkdir -p \"$HOME/bin\"\nif [[ \":$PATH:\" != *\":$HOME/bin:\"* ]]; then\n    echo \"Adding \\$HOME/bin to PATH in ~/.bashrc and exporting now...\"\n    echo 'export PATH=\"$HOME/bin:$PATH\"' >> \"$HOME/.bashrc\"\n    export PATH=\"$HOME/bin:$PATH\"\nfi\n\n# 2. Check if jq is installed\nif ! command -v jq >/dev/null 2>&1; then\n    echo \"ERROR: 'jq' is not installed. Please run: sudo apt install jq\"\n    exit 1\nfi\n\n# 3. Create jemai script\ncat <<'EOF' > \"$HOME/bin/jemai\"\n#!/bin/bash\nif [ $# -eq 0 ]; then\n    echo \"Usage: jemai <your question or command>\"\n    exit 1\nfi\ncurl -s -X POST http://localhost:11436/chat \\\n  -H \"Content-Type: application/json\" \\\n  -d \"{\\\"query\\\":\\\"$*\\\"}\" | jq -r '.response // .error // .message'\nEOF\n\nchmod +x \"$HOME/bin/jemai\"\n\n# 4. Ensure jemai is found in PATH\nif ! command -v jemai >/dev/null 2>&1; then\n    echo \"ERROR: jemai is not in your PATH! Open a new terminal or run: export PATH=\\\"$HOME/bin:\\$PATH\\\"\"\n    exit 1\nfi\n\n# 5. Test it!\necho \"Testing 'jemai ping'...\"\njemai ping\n\necho \"\u2705 jemai command is set up and working. Try:\"\necho \"   jemai what is the weather in Perth?\"\n
