[Unit]
    Description=TightVNC remote desktop service
    After=network.target

    [Service]
    Type=forking
    ExecStartPre=/usr/bin/rm -f /tmp/.X%i-lock /tmp/.X11-unix/X%i
    ExecStart=/usr/bin/tightvncserver :%i -geometry 1280x800 -depth 24 -auth %h/.Xauthority -log %h/.vnc/%H:%i.log
    ExecStop=/usr/bin/tightvncserver -kill :%i

    User=%i
    Group=%i
    WorkingDirectory=%h
    PIDFile=/home/%i/.vnc/%H:%i.pid

    [Install]
    WantedBy=multi-user.target
