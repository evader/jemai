\n# ~/.profile: executed by the command interpreter for login shells.\n\n# include .bashrc if running bash\nif [ -n \"$BASH_VERSION\" ]; then\n    if [ -f \"$HOME/.bashrc\" ]; then\n        . \"$HOME/.bashrc\"\n    fi\nfi\n\n# set PATH so it includes user's private bin if it exists\n[ -d \"$HOME/bin\" ] && PATH=\"$HOME/bin:$PATH\"\n[ -d \"$HOME/.local/bin\" ] && PATH=\"$HOME/.local/bin:$PATH\"\n\n# source any custom local environment script\n[ -f \"$HOME/.local/bin/env\" ] && . \"$HOME/.local/bin/env\"\n
