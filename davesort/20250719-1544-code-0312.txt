# --- 1. Navigate to the JEM AI Docker directory ---
cd /opt/ai_offline_setup/docker_setup

# --- 2. Stop and remove all JEM AI containers defined in docker-compose.yml ---
sudo docker compose down --remove-orphans

# --- 3. Aggressively prune ALL stopped containers, unused networks, and ALL unused volumes/images ---
# This will ask for confirmation (type 'y' and Enter).
sudo docker system prune --all --volumes

# --- 4. Restart the Docker daemon itself ---
sudo systemctl restart docker

# --- 5. Brief pause to allow Docker daemon to fully restart ---
sleep 5
