System Preparation\nsudo apt-get update\nsudo apt-get install -y python3 python3-venv python3-pip git docker.io curl lsof jq\n\n# 2. VS Code Install\nif ! command -v code &>/dev/null; then\n  sudo snap install code --classic || sudo apt-get install -y code || true\nfi\n\n# 3. Docker Install/Check\nsudo systemctl enable --now docker\n\n# 4. Check for Local Synapz Core\nif [ ! -d \"$APP_DIR\" ]; then\n  echo \"ERROR: $APP_DIR does not exist. Ollama Restore (if backup exists)\nif [ -d \"$OLLAMA_BACKUP\" ]; then\n  ollama import $OLLAMA_BACKUP/* || true\nfi\n\n# 7.
