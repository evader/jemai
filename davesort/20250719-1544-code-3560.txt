\n   import React, { useEffect, useState } from 'react';\n\n   const ModelList = () => {\n     const [models, setModels] = useState([]);\n\n     useEffect(() => {\n       fetch('/models')\n         .then(response => response.json())\n         .then(data => setModels(data))\n         .catch(error => console.error('Error fetching models:', error));\n     }, []);\n\n     return (\n       <div>\n         <h2>Ollama Models</h2>\n         <ul>\n           {models.map((model, index) => (\n             <li key={index}>{model}</li>\n           ))}\n         </ul>\n       </div>\n     );\n   };\n\n   export default ModelList;\n
