\n#!/bin/bash\n\n# Define API URL and authentication details\nAPI_URL=\"http://localhost:11436/chat\"\nUSERNAME=\"your_username\"  # Replace with your desired username\nPASSWORD=\"your_password\"  # Replace with your desired password\n\n# Step 1: Verify if Synapz Core API is running\necho \"Checking if Synapz Core API is active...\"\nSTATUS=$(curl -s -o /dev/null -w \"%{http_code}\" http://localhost:11436/status)\n\nif [ \"$STATUS\" -eq 200 ]; then\n    echo \"\u2705 Synapz Core API is active.\"\nelse\n    echo \"\u274c Synapz Core API is not responding. Please start the server first.\"\n    exit 1\nfi\n\n# Step 2: Send a test chat query\necho \"Sending a test message to /chat...\"\nQUERY=\"What is your name?\"\n\nRESPONSE=$(curl -s -u \"$USERNAME:$PASSWORD\" -X POST $API_URL -H \"Content-Type: application/json\" -d '{\"query\": \"'\"$QUERY\"'\"}')\n\n# Output the response\necho \"Response from Synapz Core: $RESPONSE\"\n
