\n# List all containers (running)\ndocker ps\n\n# List all containers (including stopped)\ndocker ps -a\n\n# List all images\ndocker images\n\n# Start/stop/restart container\ndocker start <name_or_id>\ndocker stop <name_or_id>\ndocker restart <name_or_id>\n\n# Remove container/image\ndocker rm <container_id>\ndocker rmi <image_id>\n\n# Remove ALL stopped containers, unused images, networks, build cache\ndocker system prune -a\n\n# Check logs\ndocker logs <container_name>\n\n# Get into a running container (shell)\ndocker exec -it <container_name> bash\n
