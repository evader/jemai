def create_tray_icon():
"""
Creates and runs the system tray icon. This is a blocking call.
"""
# Create a dummy image for the icon
width = 64
height = 64
color1 = (20, 20, 80) # Dark Blue
color2 = (70, 180, 255) # Light Blue
image = Image.new('RGB', (width, height), color1)
for i in range(width):
for j in range(height):
if (i - width/2)**2 + (j - height/2)**2 < (width/3)**2:
image.putpixel((i, j), color2)

### If You Get an Error (PowerShell Execution Policy)
