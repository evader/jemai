\n#!/bin/bash\n# ~/synapz_core/watchdog.sh\n\n# Check if Synapz Core is running\nif ! pgrep -f \"synapz_core.py\" > /dev/null; then\n    echo \"Synapz Core is down! Restarting...\"\n    sudo systemctl restart synapz_core\nelse\n    echo \"Synapz Core is running fine.\"\nfi\n\n# Check if Ollama is running\nif ! pgrep -f \"ollama serve\" > /dev/null; then\n    echo \"Ollama is down! Restarting...\"\n    docker start jemai_ollama\nelse\n    echo \"Ollama is running fine.\"\nfi\n
